pragma solidity ^0.8.0;
import "./FoodCommerceTypes.sol";
import "./FoodCommerceLibrary.sol";

// Logica di proprietà gestita da un contratto a parte
contract Ownable {
    address internal owner;
    constructor() {
        owner = msg.sender;
    }
    modifier onlyOwner() {
        require(msg.sender == owner, "Funzione esclusiva dell'amministratore");
        _;
    }
}

// Contratto derivato che eredita da Ownable 
contract FoodCommerceManager is Ownable {
    uint public salesSum = 0; // Variabile per il numero totale di vendite registrate 
    Product [] products; // Array della Struct dei prodotti disponibili     

    // Uso libreria esterna per usare struct condivisa vendite :
    mapping ( uint => FoodCommerceTypes.Sale ) private salesMap; 
    
    // Prodotti in vendita 
    struct Product { 
        uint idProduct;
        string nameProduct;
        uint availableProduct;
        uint priceProduct;
    }
    
    constructor() {
        // Prodotti presenti nell'e-commerce al deploy del contratto
        products.push(Product(1,"Organic Lentils", 100, 1 ether));
        products.push(Product(2,"Whole Wheat Pasta", 100, 1 ether));
        products.push(Product(3,"Plant Protein Mix", 100, 1 ether));
    }

    // Evento che traccia quando nuova vendita sarà fatta
    event Purchase ( 
        uint indexed idSale,
        uint indexed idProduct,
        string nameProduct, 
        uint amountSold,
        uint totalPaid,
        uint purchaseTimestamp, 
        address indexed buyer
        ); 

    // Verifica fondi contratto prima di autorizzare prelievo
    modifier hasSufficientFund(uint _amount) { 
        require (address(this).balance >= _amount, "Fondi insufficienti");
        _;
    }    

    // Funzione di prelievo riservata al proprietario del contratto
    function withdraw (uint _amount) public onlyOwner hasSufficientFund(_amount) {  
        (bool success, ) = payable(msg.sender).call{value: _amount}("");
        require(success, "Trasferimento fallito");
    } 

    // Funzione per aggiungere nuovo prodotto riservata al proprietario del contratto 
    function addProduct (uint _idProduct, string memory _nameProduct, uint _availableProduct, uint _priceProduct) public onlyOwner returns (string memory) {
        require(_idProduct > 0, "ID non disponibile");
        require(bytes(_nameProduct).length > 0, "Nome non valido");
        require(_availableProduct > 0, "Quantita' non valida");
        require(_priceProduct > 0, "Prezzo non valido");
        for (uint i = 0; i < products.length; i++)  {
            if(_idProduct == products[i].idProduct) {
                revert("ID gia' esistente");
            }            
        }
        products.push(Product(_idProduct, _nameProduct, _availableProduct, _priceProduct));        
        return ("Nuovo Prodotto inserito"); 
    }  

    // Funzione per leggere info prodotto inserendo Id identificativo
    function getProduct(uint _idProduct) public view returns (uint, string memory, uint, uint) {
            for (uint i = 0; i < products.length; i++) {
                if (_idProduct == products[i].idProduct) {
                    return (
                    products[i].idProduct,
                    products[i].nameProduct,
                    products[i].availableProduct,
                    products[i].priceProduct
                    );
                }
            }
             
        revert("Prodotto inesistente");
    }

    // Funzione per leggere info vendita inserendo Id identificativo
    function getSale(uint _idSale) public view returns (uint, uint, string memory, uint, uint, uint, address, uint) {
        FoodCommerceTypes.Sale memory selectedSale = salesMap[_idSale];
            if ( selectedSale.idSale ==0 ) {
                revert ("Vendita inesistente");
            }
            return (
                selectedSale.idSale,
                selectedSale.idProduct,
                selectedSale.nameProduct,
                selectedSale.amountSold,
                selectedSale.totalPaid,
                selectedSale.purchaseTimestamp,
                selectedSale.buyer,        
                selectedSale.rating            
            );
    }

    // Uso libreria per visualizzare acquisto filtrato per buyer
    function getPurchasesByBuyer(address _buyer) public view returns (FoodCommerceTypes.Sale[] memory) {
        return FoodCommerceLibrary.getPurchasesByBuyer(salesMap, salesSum, _buyer);
    }

    // Uso libreria per calcolare il totale di ETH dato un certo periodo di tempo 
    function getTotalSalesInPeriod(uint _startTimestamp, uint _endTimestamp) public view returns (uint) {
        return FoodCommerceLibrary.getTotalSalesInPeriod(salesMap, salesSum, _startTimestamp, _endTimestamp);
    }
   

    // Funzione per acquistare un prodotto
    function buyProduct( uint _idProduct, uint _amountSold ) public payable {
        for ( uint i = 0; i < products.length; i++){
            if ( _idProduct == products[i].idProduct){
                if (_amountSold == 0 ){
                    revert("Quantita' non consentita");
                }
                if (_amountSold > products[i].availableProduct ) {
                    revert("Quantita' non disponibile");
                }
                uint payAmount = products[i].priceProduct * _amountSold ;
                require(msg.value == payAmount, "Importo errato");
                products[i].availableProduct = products[i].availableProduct - _amountSold;
                salesSum++;
                uint purchaseTime = block.timestamp;
                FoodCommerceTypes.Sale memory newSale = FoodCommerceTypes.Sale(
                    salesSum,
                    products[i].idProduct,
                    products[i].nameProduct,
                    _amountSold,
                    payAmount,
                    purchaseTime,
                    msg.sender, 
                    0
                );
                salesMap[newSale.idSale] = newSale;

                emit Purchase (
                newSale.idSale,
                newSale.idProduct,
                newSale.nameProduct,
                newSale.amountSold,
                newSale.totalPaid,
                newSale.purchaseTimestamp, 
                newSale.buyer
                );
                return ;
            }
        }
        revert("Prodotto inesistente");
    }     

    // Funzione per valutazione acquisto
    function rateSale(uint _idSale, uint _rating) public {
        FoodCommerceTypes.Sale storage selectedSale = salesMap[_idSale];
        if (selectedSale.idSale == 0) {
            revert("Vendita inesistente");
        } else if (msg.sender != selectedSale.buyer) {
            revert("Solo l'acquirente puo' valutare la vendita");
        } else if (_rating < 1 || _rating > 5 ) {
            revert("Valutazione non valida");
        } else if ( selectedSale.rating != 0) {
            revert("Vendita gia' valutata");
        } else {
            selectedSale.rating = _rating;
        }         
    }
}
