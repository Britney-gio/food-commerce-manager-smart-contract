pragma solidity ^0.8.0;
import "./FoodCommerceTypes.sol";


library FoodCommerceLibrary {
    // Funzione per restituire vendite effettuate da un Buyer specifico
    function getPurchasesByBuyer(
        mapping(uint => FoodCommerceTypes.Sale) storage salesMap,
        uint salesSum,
        address _buyer
    ) internal view returns (FoodCommerceTypes.Sale[] memory) {
        uint count = 0;
        // CICLO 1 : conta quante vendite appartengono al buyer inserito
         for (uint i = 1; i <= salesSum; i++) {
            if (salesMap[i].buyer == _buyer) {
                count++;
            }
        }
        // Array dinamico di dimensione 'count'
        FoodCommerceTypes.Sale[] memory buyerSales = new FoodCommerceTypes.Sale[](count);
        uint index = 0;
        // CICLO 2 : copia vendite del buyer inserito
        for (uint i = 1; i <= salesSum; i++) {
            if (salesMap[i].buyer == _buyer) {
                buyerSales[index] = salesMap[i];
                index++;
            }
        }
        return buyerSales;
    }

    // Funzione per calcolare il totale delle vendite in un intervallo temporale
    function getTotalSalesInPeriod(
        mapping(uint => FoodCommerceTypes.Sale) storage salesMap,
        uint salesSum,
        uint _startTimestamp,
        uint _endTimestamp
        ) internal view returns (uint) {
        require(_startTimestamp <= _endTimestamp, "Intervallo di tempo non valido");
        uint totalAmount = 0;
        for (uint i = 1; i <= salesSum; i++) {
            if (
                salesMap[i].purchaseTimestamp >= _startTimestamp &&
                salesMap[i].purchaseTimestamp <= _endTimestamp
            ) {
                totalAmount += salesMap[i].totalPaid;
            }
        }
        return totalAmount;
    }
}
